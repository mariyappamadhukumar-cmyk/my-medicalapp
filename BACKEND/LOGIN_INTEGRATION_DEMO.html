💻 How It Works (Code)

Check Authentication Status:

function checkAuthStatus() { const token = localStorage.getItem('authToken'); const userInfo = JSON.parse(localStorage.getItem('userInfo') || '{}'); if (token && userInfo.name) { // User is logged in - show greeting authButton.innerHTML = '🚪 Logout'; userGreeting.style.display = 'flex'; userName.textContent = userInfo.name; userAvatar.textContent = userInfo.name.charAt(0).toUpperCase(); } else { // User is not logged in - show login button authButton.innerHTML = '🔐 Login'; userGreeting.style.display = 'none'; } }

Handle Login/Logout:

function handleAuth() { const token = localStorage.getItem('authToken'); if (token) { // Logout if (confirm('Are you sure you want to logout?')) { localStorage.removeItem('authToken'); localStorage.removeItem('userInfo'); checkAuthStatus(); } } else { // Go to login page window.location.href = 'login.html'; } }
